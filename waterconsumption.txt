import java.util.Scanner;
class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int A= sc.nextInt();
        for (int i = 0; i < A; i++) {
            int X = sc.nextInt();
            if (X >= 2000) {
                System.out.println("YES");
            } else {
                System.out.println("NO");
            }
        }
    }
}
